JavaScript

Guía rápida de JavaScript moderno: variables, tipos, template strings, operadores, control de flujo, funciones, arrays/strings/números, módulos, DOM, asincronía (promesas/fetch), fechas, storage y objetos/JSON.

Variables

Usa const por defecto. let si cambia. Evita var. 

// var (EVITAR)
var nombre = "Juan";
var nombre = "Pedro";   // redeclarar ✅
nombre = "Carlos";      // reasignar ✅

// Scope: función / global (NO bloque)
if (true) {
  var mensaje = "Hola desde var";
}
console.log(mensaje);   // ✅ funciona (var ignora bloque)

// let (cambia)
let edad = 30;
edad = 35;              // reasignar ✅
// let edad = 40;       // ❌ redeclarar en el mismo bloque

// Scope: bloque
if (true) {
  let saludo = "Hola desde let";
  console.log(saludo);
}
// console.log(saludo); // ❌ fuera del bloque

// const (fijo / referencia estable)
const PI = 3.1416;
// PI = 3.15;           // ❌ no reasignar
  • const no “congela” objetos/arrays, solo evita reasignar la variable.

Tipos de datos

Primitivos (valor) vs complejos (referencia) + typeof. 

// Primitivos
const texto = "JavaScript";   // string
const entero = 42;            // number
const esActivo = true;        // boolean
let indefinido;               // undefined
const vacio = null;           // null
const simbolo = Symbol("id"); // symbol
const grande = BigInt(1);     // bigint

// Complejos
const persona = { nombre: "Ana", edad: 25 };
const frutas = ["🍎", "🍌"];
function saludar(n) { return `Hola ${n}`; }

// Copia por valor (primitivo)
let a = 5;
let b = a;
b = 10;
console.log(a); // 5

// Copia por referencia (objeto)
let obj1 = { nombre: "Juan" };
let obj2 = obj1;
obj2.nombre = "Pedro";
console.log(obj1.nombre); // "Pedro"

// typeof
console.log(typeof "Hola");      // "string"
console.log(typeof 42);          // "number"
console.log(typeof null);        // "object" (histórico)
console.log(typeof []);          // "object"
console.log(typeof function(){});// "function"

Operadores

Aritmética, asignación, comparación, lógica y modernos. 

// Aritméticos
5 + 3;
10 - 4;
3 * 4;
20 / 4;
10 % 3;
2 ** 3;

// Comparación
5 == "5";    // true (EVITAR)
5 === "5";   // false (RECOMENDADO)
5 !== "5";   // true

// Lógicos
true && false;
true || false;
!true;

// Ternario
const edad = 18;
const acceso = edad >= 18 ? "Permitido" : "Denegado";

// Modernos
const nombre = null;
console.log(nombre ?? "Anónimo"); // nullish coalescing

const usuario = { direccion: { ciudad: "Madrid" } };
console.log(usuario.direccion?.ciudad);
console.log(usuario.contacto?.telefono); // undefined sin error
  • Regla práctica: usa === y !== casi siempre.

Estructuras de control

Bucles + condicionales + break/continue/return. 

// for
for (let i = 0; i < 3; i++) console.log(i);

// while
let c = 0;
while (c < 2) { c++; }

// do...while
let intentos = 0;
do { intentos++; } while (intentos < 1);

// for...of (arrays)
for (const fruta of ["🍎","🍌"]) console.log(fruta);

// for...in (objetos)
const persona = { nombre: "Ana", edad: 25 };
for (const k in persona) console.log(k, persona[k]);

// if / else
const edad = 20;
if (edad >= 18) console.log("OK");
else console.log("NO");

// switch
const dia = "Lunes";
switch (dia) {
  case "Lunes": console.log("Inicio"); break;
  default: console.log("Normal");
}

// break / continue
for (let i = 0; i < 5; i++) {
  if (i === 2) continue;
  if (i === 4) break;
  console.log(i);
}

Funciones

Declaradas, expresadas, arrow, orden superior, defaults y recursión. 

// Declarada
function saludar(nombre) {
  return `Hola, ${nombre}`;
}

// Expresada
const despedir = function (nombre) {
  return `Adiós, ${nombre}`;
};

// Arrow
const sumar = (a, b) => a + b;

// Arrow con bloque
const multiplicar = (a, b) => {
  const r = a * b;
  return r;
};

// Orden superior
function operar(a, b, op) {
  return op(a, b);
}
const res = operar(4, 2, (x, y) => x * y);

// Defaults
function saludarPersona(nombre = "Anónimo") {
  console.log(`Hola, ${nombre}`);
}

// Recursiva
function cuentaRegresiva(n) {
  if (n === 0) return console.log("BOOM");
  console.log(n);
  cuentaRegresiva(n - 1);
}
  • Arrow function: no tiene su propio this (importante en objetos/clases).

Métodos de Array

Mutables vs inmutables (recomendado) + iteración. 

// Mutan el array
const nums = [1,2,3];
nums.push(4);
nums.pop();
nums.shift();
nums.unshift(0);
nums.splice(1, 1, 99);
nums.reverse();
["b","a","c"].sort();

// NO mutan (preferidos)
const a = [1,2,3,4,5];

const cuadrados = a.map(n => n ** 2);
const pares = a.filter(n => n % 2 === 0);
const suma = a.reduce((acc, n) => acc + n, 0);
const encontrado = a.find(n => n > 3);
const hayPares = a.some(n => n % 2 === 0);
const todosPos = a.every(n => n > 0);

a.includes(3);
a.indexOf(3);
a.slice(1, 3);
a.concat([6,7]);
a.join(" - ");

a.forEach(n => console.log(n));
  • Regla práctica: map/filter/reduce para transformar; evita mutar si no hace falta.

Módulos (ESM)

Export / import (tu proyecto ya usa type="module"). 

// miModulo.js
export const saludar = (nombre) => `Hola, ${nombre}`;

const mensaje = "Bienvenido";
export default mensaje;

// app.js
import mensaje, { saludar } from "./miModulo.js";

// Importar todo
import * as miModulo from "./miModulo.js";

Promesas

then/catch/finally + all/race.

const promesa = new Promise((resolve, reject) => {
  const ok = true;
  if (ok) resolve("✅ OK");
  else reject("❌ Error");
});

promesa
  .then(r => console.log(r))
  .catch(e => console.error(e))
  .finally(() => console.log("Fin"));

// Múltiples
const p1 = Promise.resolve("Primera");
const p2 = Promise.resolve("Segunda");

Promise.all([p1, p2]).then(console.log);   // ["Primera","Segunda"]
Promise.race([p1, p2]).then(console.log);  // "Primera"

Fetch / Async-Await

GET/POST + manejo de errores con response.ok. 

// Fetch (then)
fetch("https://jsonplaceholder.typicode.com/posts/1")
  .then(res => {
    if (!res.ok) throw new Error("HTTP Error");
    return res.json();
  })
  .then(data => console.log(data))
  .catch(err => console.error(err));

// Fetch (POST)
fetch("https://jsonplaceholder.typicode.com/posts", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ title: "Nuevo", body: "Texto" }),
})
  .then(res => res.json())
  .then(console.log)
  .catch(console.error);

// Async/Await
async function obtenerPokemon(id) {
  try {
    const res = await fetch(`https://pokeapi.co/api/v2/pokemon/${id}`);
    if (!res.ok) throw new Error(res.status);
    const data = await res.json();
    return data;
  } catch (e) {
    console.error("Error:", e);
  } finally {
    console.log("Fin");
  }
}
  • fetch NO lanza error por 404/500: revisa res.ok.

Storage (browser)

localStorage, sessionStorage y cookies.

// localStorage (persistente)
localStorage.setItem("nombre", "Juan");
localStorage.setItem("config", JSON.stringify({ tema: "oscuro" }));

const nombre = localStorage.getItem("nombre");
const config = JSON.parse(localStorage.getItem("config"));

localStorage.removeItem("nombre");
localStorage.clear();

// sessionStorage (pestaña)
sessionStorage.setItem("usuario", "Ana");
sessionStorage.getItem("usuario");
sessionStorage.clear();

// Cookies (simple)
document.cookie = "idioma=es; expires=Fri, 31 Dec 2030 23:59:59 GMT; path=/";

function getCookie(nombre) {
  const valor = `; ${document.cookie}`;
  const partes = valor.split(`; ${nombre}=`);
  if (partes.length === 2) return partes.pop().split(";").shift();
}
  • localStorage: fácil, pero no guardes secretos (tokens sensibles) ahí.

Resumen

  • const por defecto, let si cambia, evita var.
  • Usa ===, funciones pequeñas y arrays inmutables (map/filter/reduce).
  • DOM: querySelector, classList, eventos con addEventListener.
  • Async: async/await + try/catch + response.ok.
  • Persistencia: localStorage (no secretos), sessionStorage (pestaña), cookies (más control).
